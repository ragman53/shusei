       fn render_with_retry(
           &self,
           document: &PdfDocument,
           page_index: u32,
           width: u32,
           height: u32,
       ) -> Result<Vec<u8>> {
           // First attempt
           match self.render_page(document, page_index, width, height) {
               Ok(result) => Ok(result),
               Err(e) => {
                   log::warn!("Render failed for page {}, retrying...", page_index);
                   // Retry once
                   self.render_page(document, page_index, width, height)
               }
           }
       }
